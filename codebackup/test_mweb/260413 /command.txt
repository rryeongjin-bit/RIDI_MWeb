order →  전체 실행 시 순서 보장

#!/bin/bash
# start_appium.sh
# Appium 서버 자동 실행 + 프록시 설정 자동화

#   프록시만 설정              : bash start_appium.sh proxy
#   전체 종료 + 프록시 해제    : bash start_appium.sh stop
